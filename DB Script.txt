CREATE TABLE guest (
    national_id INT PRIMARY KEY,
    full_name VARCHAR2(50) NOT NULL,
    address VARCHAR2(200),
    nationality VARCHAR2(20) NOT NULL,
    date_of_birth DATE NOT NULL,
    gender CHAR(1) NOT NULL
);

CREATE TABLE staff (
    staff_id INT PRIMARY KEY,
    full_name VARCHAR2(50) NOT NULL,
    staff_position VARCHAR2(50) NOT NULL,
    salary DECIMAL(10, 2) NOT NULL,
    contact_info VARCHAR2(50) NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE
);

CREATE TABLE room (
    room_id INT PRIMARY KEY,
    room_type VARCHAR2(50) NOT NULL,
    price DECIMAL(10, 2) NOT NULL,
    status VARCHAR2(10) NOT NULL,
    facade VARCHAR2(50),
    staff_id INT NOT NULL,
    FOREIGN KEY (staff_id) REFERENCES staff(staff_id)
);

CREATE TABLE payment (
    payment_id INT PRIMARY KEY,
    amount DECIMAL(10, 2) NOT NULL,
    payment_date DATE NOT NULL,
    payment_method VARCHAR2(10) NOT NULL,
    national_id INT NOT NULL,
    FOREIGN KEY (national_id) REFERENCES guest(national_id) ON DELETE CASCADE
);

CREATE TABLE reservation (
    reservation_id INT PRIMARY KEY,
    check_in_date DATE NOT NULL,
    check_out_date DATE NOT NULL,
    number_of_adults INT NOT NULL,
    number_of_children INT NOT NULL,
    status VARCHAR2(20) NOT NULL,
    room_id INT NOT NULL,
    staff_id INT NOT NULL,
    payment_id INT NOT NULL,
    FOREIGN KEY (room_id) REFERENCES room(room_id),
    FOREIGN KEY (staff_id) REFERENCES staff(staff_id),
    FOREIGN KEY (payment_id) REFERENCES payment(payment_id)
);

CREATE TABLE supply (
    supply_id INT PRIMARY KEY,
    supply_type VARCHAR2(50) NOT NULL
);

CREATE TABLE dining_reservation (
    dining_reservation_id INT PRIMARY KEY,
    number_of_guests INT NOT NULL,
    dining_time DATE NOT NULL,
    national_id INT NOT NULL,
    staff_id INT NOT NULL,
    payment_id INT NOT NULL,
    FOREIGN KEY (national_id) REFERENCES guest(national_id) ON DELETE CASCADE,
    FOREIGN KEY (staff_id) REFERENCES staff(staff_id),
    FOREIGN KEY (payment_id) REFERENCES payment(payment_id)
);

CREATE TABLE phone (
    phone_number VARCHAR2(20) PRIMARY KEY,
    country_code VARCHAR2(10) NOT NULL,
    national_id INT NOT NULL,
    FOREIGN KEY (national_id) REFERENCES guest(national_id) ON DELETE CASCADE
);

CREATE TABLE email (
    email_address VARCHAR2(50) PRIMARY KEY,
    national_id INT NOT NULL,
    FOREIGN KEY (national_id) REFERENCES guest(national_id) ON DELETE CASCADE
);

CREATE TABLE dining_order (
    order_id INT PRIMARY KEY,
    order_time DATE NOT NULL,
    status VARCHAR2(10) NOT NULL,
    price DECIMAL(10, 2) NOT NULL,
    dining_reservation_id INT NOT NULL,
    FOREIGN KEY (dining_reservation_id) REFERENCES dining_reservation(dining_reservation_id) ON DELETE CASCADE
);

CREATE TABLE menu_item (
    item_id INT PRIMARY KEY,
    item_name VARCHAR2(50) NOT NULL,
    item_category VARCHAR2(20) NOT NULL,
    item_description VARCHAR2(100) NOT NULL,
    price DECIMAL(10, 2) NOT NULL
);

CREATE TABLE guest_reservation (
    national_id INT NOT NULL,
    reservation_id INT NOT NULL,
    PRIMARY KEY (national_id, reservation_id),
    FOREIGN KEY (national_id) REFERENCES guest(national_id) ON DELETE CASCADE,
    FOREIGN KEY (reservation_id) REFERENCES reservation(reservation_id) ON DELETE CASCADE
);

CREATE TABLE supply_room (
    supply_id INT NOT NULL,
    room_id INT NOT NULL,
    quantity INT NOT NULL,
    PRIMARY KEY (supply_id, room_id),
    FOREIGN KEY (supply_id) REFERENCES supply(supply_id) ON DELETE CASCADE,
    FOREIGN KEY (room_id) REFERENCES room(room_id) ON DELETE CASCADE
);

CREATE TABLE staff_supply (
    staff_id INT NOT NULL,
    supply_id INT NOT NULL,
    PRIMARY KEY (staff_id, supply_id),
    FOREIGN KEY (staff_id) REFERENCES staff(staff_id) ON DELETE CASCADE,
    FOREIGN KEY (supply_id) REFERENCES supply(supply_id) ON DELETE CASCADE
);

CREATE TABLE order_menu_item (
    order_id INT NOT NULL,
    item_id INT NOT NULL,
    PRIMARY KEY (order_id, item_id),
    FOREIGN KEY (order_id) REFERENCES dining_order(order_id) ON DELETE CASCADE,
    FOREIGN KEY (item_id) REFERENCES menu_item(item_id) ON DELETE CASCADE
);

CREATE OR REPLACE TRIGGER trg_check_availability
BEFORE INSERT ON reservation
FOR EACH ROW
DECLARE
    v_count INT;
BEGIN
    IF :NEW.check_in_date < TRUNC(SYSDATE) THEN
        RAISE_APPLICATION_ERROR(-20003, 'The check-in date cannot be in the past.');
    ELSE
        SELECT COUNT(*)
        INTO v_count
        FROM reservation
        WHERE room_id = :NEW.room_id
            AND (:NEW.check_in_date < check_out_date AND :NEW.check_out_date > check_in_date);
    
        IF v_count > 0 THEN
            RAISE_APPLICATION_ERROR(-20001, 'This room is already booked for the given dates.');
        END IF;
    END IF;
END;
/

CREATE OR REPLACE TRIGGER trg_update_room_status
AFTER INSERT OR UPDATE OF status ON reservation
FOR EACH ROW
WHEN (NEW.status = 'checked-in')
BEGIN
  UPDATE room
  SET status = 'Occupied'
  WHERE room_id = :NEW.room_id;
END;
